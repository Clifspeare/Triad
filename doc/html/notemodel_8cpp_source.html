90 QVariant NoteModel::data(const QModelIndex &index, int role) const
91 {
92  if (!index.isValid())
93  return QVariant();
94 
95  if (role == NoteModelRoles::Title) {
96  DomItem *item = static_cast<DomItem*>(index.internalPointer());
97  QDomNode node = item->node();
98  QDomNamedNodeMap atts = node.attributes();
99  if (atts.contains("title")) {
100  QDomAttr attr = atts.namedItem("title").toAttr();
101  return attr.value();
102  } else {
103  return QVariant();
104  }
105  } else if (role == NoteModelRoles::Content) {
106  DomItem *item = static_cast<DomItem*>(index.internalPointer());
107  QDomNode node = item->node();
108  return QVariant::fromValue(node);
109  } else if (role == NoteModelRoles::XPOSITION) {
110  DomItem *item = static_cast<DomItem*>(index.internalPointer());
111  QDomNode node = item->node();
112  QDomNamedNodeMap atts = node.attributes();
113  if (node.nodeName() == "ContentContainer") {
114  if (atts.contains("XPos")) { //all ContentContainer nodes should have an XPos attribute, this is just in case.
115  QDomAttr attr = atts.namedItem("XPos").toAttr();
116  bool errChk;
117  QVariant xValue = attr.value().toDouble(&errChk);
118  if (errChk == false) { //if value stored at XPos is for some reason not convertible to a double.
119  qDebug() << "Conversion error occurred: " << __LINE__ << " in " << __FILE__;
120  return QVariant();
121  } else {
122  return xValue;
123  }
124  } else {
125  qDebug() << "Error parsing xml document.";
126  return QVariant();
127  }
128  }
129  } else if (role == NoteModelRoles::YPOSITION) {
130  DomItem *item = static_cast<DomItem*>(index.internalPointer());
131  QDomNode node = item->node();
132  QDomNamedNodeMap atts = node.attributes();
133  if (node.nodeName() == "ContentContainer") {
134  if (atts.contains("YPos")) { //all ContentContainer nodes should have an XPos attribute, this is just in case.
135  QDomAttr attr = atts.namedItem("YPos").toAttr();
136  bool errChk;
137  QVariant xValue = attr.value().toDouble(&errChk);
138  if (errChk == false) { //if value stored at XPos is for some reason not convertible to a double.
139  qDebug() << "Conversion error occurred: " << __LINE__ << " in " << __FILE__;
140  return QVariant();
141  } else {
142  return xValue;
143  }
144  } else {
145  qDebug() << "Error parsing xml document.";
146  return QVariant();
147  }
148  }
149  }
150  return QVariant();
151 }

162 {

166  DomItem *parentItem;

180 QModelIndex NoteModel::parent(const QModelIndex &child) const
181 {

184 
185  DomItem *childItem = static_cast<DomItem*>(child.internalPointer());
186  DomItem *parentItem = childItem->parent();
